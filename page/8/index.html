#!/bin/bash

if [ $# == 0 ] ; then
    echo "Usage: lara-test [suite] [filter]"
    exit
fi

cd `pwd`

COMMAND="php artisan test"
SUITE=$1
FILTER=$2

if [ $# == 1 ] ; then
    $COMMAND --testsuite $SUITE
    exit
fi

if [ $# == 2 ] ; then
    $COMMAND --testsuite $SUITE --filter $FILTER
    exit
fi

shift
shift
$COMMAND --testsuite $SUITE --filter $FILTER $@
